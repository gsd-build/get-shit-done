<!-- react-contenteditable pattern from RESEARCH.md -->

```typescript
import ContentEditable from 'react-contenteditable';
import { useRef, useCallback } from 'react';

function InlineEditor({ value, onChange, disabled }) {
  const contentRef = useRef(value);
  const originalRef = useRef(value);

  const handleChange = useCallback((evt) => {
    contentRef.current = evt.target.value;
  }, []);

  const handleBlur = useCallback(() => {
    if (contentRef.current !== originalRef.current) {
      onChange(contentRef.current, originalRef.current);
      originalRef.current = contentRef.current;
    }
  }, [onChange]);

  return (
    <ContentEditable
      html={value}
      onChange={handleChange}
      onBlur={handleBlur}
      disabled={disabled}
    />
  );
}
```
