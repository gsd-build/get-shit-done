1. Create packages/gsd-wrapper/src/exec.ts - shared subprocess execution utility:
```typescript
import { spawn } from 'child_process';
import type { GsdError, GsdResult } from './types.js';

// Path to gsd-tools.cjs relative to monorepo root
const GSD_TOOLS_PATH = new URL('../../../get-shit-done/bin/gsd-tools.cjs', import.meta.url).pathname;

export async function execGsdTools<T>(
  args: string[],
  cwd: string
): Promise<GsdResult<T>> {
  return new Promise((resolve) => {
    const proc = spawn('node', [GSD_TOOLS_PATH, ...args], {
      cwd,
      stdio: ['ignore', 'pipe', 'pipe'],
    });

    let stdout = '';
    let stderr = '';

    proc.stdout.on('data', (data) => { stdout += data; });
    proc.stderr.on('data', (data) => { stderr += data; });

    proc.on('close', (code) => {
      if (code !== 0) {
        resolve({
          success: false,
          error: {
            code: 'GSD_COMMAND_FAILED',
            message: `gsd-tools ${args[0]} failed with code ${code}`,
            command: `gsd-tools ${args.join(' ')}`,
            stderr: stderr.trim() || undefined,
          },
        });
        return;
      }

      try {
        const data = JSON.parse(stdout) as T;
        resolve({ success: true, data });
      } catch {
        resolve({
          success: false,
          error: {
            code: 'GSD_PARSE_ERROR',
            message: 'Failed to parse gsd-tools JSON output',
            command: `gsd-tools ${args.join(' ')}`,
            stderr: stdout.substring(0, 200),
          },
        });
      }
    });

    proc.on('error', (err) => {
      resolve({
        success: false,
        error: {
          code: 'GSD_SPAWN_ERROR',
          message: err.message,
          command: `gsd-tools ${args.join(' ')}`,
        },
      });
    });
  });
}
```

```typescript
import { execGsdTools } from './exec.js';
import type { ProjectState, GsdResult } from './types.js';

// Raw output from gsd-tools state
interface RawState {
  phase: string;
  plan: string;
  status: string;
  progress?: { total: number; completed: number };
  lastActivity?: string;
  milestone?: string;
}

export async function getState(cwd: string): Promise<GsdResult<ProjectState>> {
  const result = await execGsdTools<RawState>(['state', '--json'], cwd);

  if (!result.success) return result;

  const raw = result.data;
  const total = raw.progress?.total ?? 0;
  const completed = raw.progress?.completed ?? 0;

  return {
    success: true,
    data: {
      phase: raw.phase,
      plan: raw.plan,
      status: (raw.status as ProjectState['status']) || 'pending',
      progress: {
        total,
        completed,
        percentage: total > 0 ? Math.round((completed / total) * 100) : 0,
      },
      lastActivity: raw.lastActivity,
      milestone: raw.milestone,
    },
  };
}
```

```typescript
import { execGsdTools } from './exec.js';
import type { Phase, GsdResult } from './types.js';

interface RawPhasesOutput {
  directories: string[];
  count: number;
  milestone?: string;
}

export async function listPhases(cwd: string): Promise<GsdResult<Phase[]>> {
  const result = await execGsdTools<RawPhasesOutput>(['phases', 'list', '--json'], cwd);

  if (!result.success) return result;

  // Parse directory names into Phase objects
  const phases: Phase[] = result.data.directories.map((dir) => {
    // Directory format: NN-name or NN-name-slug
    const basename = dir.split('/').pop() || dir;
    const match = basename.match(/^(\d+(?:\.\d+)?)-(.+)$/);

    if (!match) {
      return {
        number: '0',
        name: basename,
        slug: basename,
        status: 'pending' as const,
        plans: 0,
        completedPlans: 0,
        directory: dir,
      };
    }

    const [, number, rest] = match;
    return {
      number: number || '0',
      name: rest?.replace(/-/g, ' ') || '',
      slug: rest || '',
      status: 'pending' as const, // Would need to read PLAN.md files for actual status
      plans: 0,
      completedPlans: 0,
      directory: dir,
    };
  });

  return { success: true, data: phases };
}
```

```typescript
import { readdir, access, constants } from 'fs/promises';
import { join, basename } from 'path';
import type { Project, GsdResult } from './types.js';

/**
 * Discover GSD projects by scanning for .planning directories
 */
export async function discoverProjects(
  searchPaths: string[]
): Promise<GsdResult<Project[]>> {
  const projects: Project[] = [];

  for (const searchPath of searchPaths) {
    try {
      const entries = await readdir(searchPath, { withFileTypes: true });

      for (const entry of entries) {
        if (!entry.isDirectory()) continue;

        const projectPath = join(searchPath, entry.name);
        const planningPath = join(projectPath, '.planning');

        try {
          await access(planningPath, constants.R_OK);
          projects.push({
            id: entry.name,
            name: entry.name,
            path: projectPath,
            hasPlanning: true,
          });
        } catch {
          // No .planning directory - not a GSD project
        }
      }
    } catch {
      // Search path not accessible - skip
    }
  }

  return { success: true, data: projects };
}
```
