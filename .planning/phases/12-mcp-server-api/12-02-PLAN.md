5. `registerCoreTools(server)` function that registers:

   a. **progress** tool:
      - description: "Show project progress across phases and milestones"
      - inputSchema: { format: z.enum(['json', 'table', 'bar']).optional(), milestone: z.string().optional() }
      - handler: calls commands.cmdProgress(cwd, format || 'json', milestone, true)
      - next_actions: ['health', 'phase_info']

   b. **health** tool:
      - description: "Check .planning/ integrity and worktree status"
      - inputSchema: { repair: z.boolean().optional().default(false) }
      - handler: calls health.cmdHealthCheck(cwd, { repair }, () => {}, true)
      - next_actions: ['progress']

   c. **state_get** tool:
      - description: "Get STATE.md content or specific section"
      - inputSchema: { section: z.string().optional() }
      - handler: calls state.cmdStateGet(cwd, section, true) or state.cmdStateSnapshot(cwd, true)
      - next_actions: ['roadmap_get', 'progress']

   d. **phase_info** tool:
      - description: "Get information about a specific phase"
      - inputSchema: { phase: z.string().describe('Phase number or M<milestone>/<phase> reference') }
      - handler: calls phase.cmdFindPhase(cwd, phaseArg, {}, true)
      - next_actions: ['roadmap_get', 'state_get']

   e. **roadmap_get** tool:
      - description: "Get ROADMAP.md parsed as structured data"
      - inputSchema: { phase: z.string().optional() }
      - handler: calls roadmap.cmdRoadmapGetPhase or roadmap.cmdRoadmapAnalyze
      - next_actions: ['phase_info', 'progress']

Each tool handler should:
- Use process.cwd() for cwd
- Wrap in try/catch
- Return envelope(data, nextActions) on success
- Return errorEnvelope(...) on error
- Use `content: [{ type: 'text', text: JSON.stringify(envelope) }]` format per MCP SDK

Per CONTEXT.md: Direct parameter mapping - MCP tool params mirror CLI flags exactly.

Update `get-shit-done/bin/gsd-mcp-server.cjs` to register tools:

```javascript
#!/usr/bin/env node

const { createServer, connectTransport } = require('./lib/mcp/server.cjs');
const { registerCoreTools, registerExtendedTools } = require('./lib/mcp/tools.cjs');

async function main() {
  const server = createServer();

  // Register core tier (always available)
  registerCoreTools(server);

  // Register extended tier
  // Per CONTEXT.md: "extended tier via capability negotiation"
  // For MVP, always register extended tools (same trust model as CLI)
  registerExtendedTools(server);

  // Resources will be registered in plan 12-03

  await connectTransport(server);

  console.error('[gsd-mcp] Server started with',
    Object.keys(server._tools || {}).length, 'tools');
}

main().catch((err) => {
  console.error('[gsd-mcp] Fatal error:', err.message);
  process.exit(1);
});
```

Note: The server._tools access may vary by SDK version. Adjust to use public API if available.
