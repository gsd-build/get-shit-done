```typescript
import { test, expect } from '@playwright/test';

test.describe('Roadmap Visualization', () => {
  // ROAD-01: Dependency graph showing phase relationships
  test('displays dependency graph with connected phase nodes', async ({ page }) => {
    await page.goto('/projects/test-project/roadmap');
    // Check for React Flow container
    await expect(page.locator('.react-flow')).toBeVisible();
    // Check for phase nodes
    await expect(page.getByText(/Phase \d+/)).toBeVisible();
    // Check for edges (React Flow renders these as SVG paths)
  });

  // ROAD-02: Gantt-style timeline
  test('displays timeline view with phase bars', async ({ page }) => {
    await page.goto('/projects/test-project/roadmap');
    // Switch to timeline view
    await page.getByRole('tab', { name: 'Timeline' }).click();
    // Check for timeline bars
    await expect(page.locator('[role="button"]').first()).toBeVisible();
  });

  // ROAD-03: Progress tracking per phase
  test('shows progress indicators on phase nodes', async ({ page }) => {
    await page.goto('/projects/test-project/roadmap');
    // Check for progress bars within nodes
    await expect(page.locator('[role="progressbar"]').first()).toBeVisible();
    // Check for status badges
    await expect(page.locator('text=In Progress').or(page.locator('text=Complete'))).toBeVisible();
  });

  // ROAD-04: Phases grouped by milestone
  test('groups phases by milestone in timeline', async ({ page }) => {
    await page.goto('/projects/test-project/roadmap');
    await page.getByRole('tab', { name: 'Timeline' }).click();
    // Check for milestone headers
    await expect(page.getByText(/v\d+\.\d+/)).toBeVisible();
  });

  // ROAD-05: Click phase to navigate to detail
  test('navigates to phase detail when clicking phase', async ({ page }) => {
    await page.goto('/projects/test-project/roadmap');
    // Click a phase node
    await page.locator('.react-flow__node').first().click();
    // Should navigate to phase detail
    await expect(page).toHaveURL(/\/projects\/test-project\/phases\/\d+/);
  });

  // Additional tests per CONTEXT.md
  test('can toggle between graph orientations', async ({ page }) => {
    await page.goto('/projects/test-project/roadmap');
    // Check for layout controls
    await page.getByRole('button', { name: /Left-Right|Top-Bottom/i }).click();
  });

  test('renders minimap for large roadmaps', async ({ page }) => {
    await page.goto('/projects/test-project/roadmap');
    // React Flow MiniMap has specific class
    await expect(page.locator('.react-flow__minimap')).toBeVisible();
  });

  test('keyboard navigation works', async ({ page }) => {
    await page.goto('/projects/test-project/roadmap');
    // Tab to focus on graph
    await page.keyboard.press('Tab');
    // Arrow key to select node
    await page.keyboard.press('ArrowDown');
    // Enter to navigate
    await page.keyboard.press('Enter');
    // Should navigate or show selection
  });
});
```
