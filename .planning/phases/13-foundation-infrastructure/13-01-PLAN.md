    Create typed event definitions for Socket.IO communication per CONTEXT.md:

    1. **packages/events/src/types.ts** - TypeScript interfaces:
       ```typescript
       // Prefixed event names per CONTEXT.md
       export const EVENTS = {
         AGENT_TOKEN: 'agent:token',
         AGENT_START: 'agent:start',
         AGENT_END: 'agent:end',
         AGENT_ERROR: 'agent:error',
         CHECKPOINT_REQUEST: 'checkpoint:request',
         CHECKPOINT_RESPONSE: 'checkpoint:response',
         CONNECTION_HEALTH: 'connection:health',
       } as const;

       // Event payloads
       export interface TokenEvent {
         agentId: string;
         token: string;
         sequence: number;
       }

       export interface AgentStartEvent {
         agentId: string;
         planId: string;
         taskName: string;
       }

       export interface AgentEndEvent {
         agentId: string;
         status: 'success' | 'error' | 'checkpoint';
         summary?: string;
       }

       export interface CheckpointRequestEvent {
         checkpointId: string;
         type: 'human-verify' | 'decision' | 'human-action';
         prompt: string;
         options?: { id: string; label: string }[];
         timeoutMs?: number;
       }

       export interface CheckpointResponseEvent {
         checkpointId: string;
         response: string;
       }

       // Socket.IO typed events
       export interface ServerToClientEvents {
         [EVENTS.AGENT_TOKEN]: (event: TokenEvent) => void;
         [EVENTS.AGENT_START]: (event: AgentStartEvent) => void;
         [EVENTS.AGENT_END]: (event: AgentEndEvent) => void;
         [EVENTS.CHECKPOINT_REQUEST]: (event: CheckpointRequestEvent) => void;
       }

       export interface ClientToServerEvents {
         [EVENTS.CHECKPOINT_RESPONSE]: (event: CheckpointResponseEvent) => void;
         'agent:subscribe': (agentId: string) => void;
       }
       ```

       ```typescript
       export const CheckpointResponseSchema = z.object({
         checkpointId: z.string().uuid(),
         response: z.string().min(1),
       });

       ```
