       ```typescript
       import { createServer } from 'http';
       import type { ServerToClientEvents, ClientToServerEvents } from '@gsd/events';

       interface ServerConfig {
         port: number;
         maxDisconnectionDuration?: number;  // Default 2 minutes per CONTEXT.md
         pingInterval?: number;               // Default 25000
         pingTimeout?: number;                // Default 20000
       }

       export function createSocketServer(config: ServerConfig) {
         const httpServer = createServer();

         const io = new Server<ClientToServerEvents, ServerToClientEvents>(httpServer, {
           connectionStateRecovery: {
             maxDisconnectionDuration: config.maxDisconnectionDuration ?? 2 * 60 * 1000,
             skipMiddlewares: true,
           },
           pingInterval: config.pingInterval ?? 25000,
           pingTimeout: config.pingTimeout ?? 20000,
           cors: {
             origin: process.env.CORS_ORIGIN ?? 'http://localhost:3000',
             credentials: true,
           },
         });

         return { httpServer, io };
       }
       ```

       ```typescript
       import { EVENTS, type ServerToClientEvents, type ClientToServerEvents } from '@gsd/events';
       import { joinProjectRoom, joinAgentRoom } from './rooms';

       export function registerHandlers(
         io: Server<ClientToServerEvents, ServerToClientEvents>
       ): void {
         io.on('connection', (socket) => {
           // Log recovery status per RESEARCH.md pattern
           if (socket.recovered) {
             console.log(`[socket] Recovered session: ${socket.id}`);
           } else {
             console.log(`[socket] New connection: ${socket.id}`);
           }

           // Join project room from handshake query
           const projectId = socket.handshake.query.projectId;
           if (typeof projectId === 'string') {
             joinProjectRoom(socket, projectId);
           }

           // Subscribe to agent streams
           socket.on('agent:subscribe', (agentId) => {
             joinAgentRoom(socket, agentId);
             console.log(`[socket] ${socket.id} subscribed to agent:${agentId}`);
           });

           // Checkpoint response handling (to be wired to agent in Phase 14)
           socket.on(EVENTS.CHECKPOINT_RESPONSE, (event) => {
             console.log(`[socket] Checkpoint response: ${event.checkpointId}`);
             // Will emit to agent orchestrator in Phase 14
           });

           socket.on('disconnect', (reason) => {
             console.log(`[socket] Disconnected: ${socket.id}, reason: ${reason}`);
           });
         });
       }
       ```

       ```typescript
       import { io, Socket } from 'socket.io-client';
       import type { ServerToClientEvents, ClientToServerEvents, TokenEvent } from './types';

       export type TypedSocket = Socket<ServerToClientEvents, ClientToServerEvents>;

       interface SocketClientConfig {
         url: string;
         projectId?: string;
         autoConnect?: boolean;
       }

       /**
        * Create typed Socket.IO client with auto-reconnection.
        */
       export function createSocketClient(config: SocketClientConfig): TypedSocket {
         const { url, projectId, autoConnect = true } = config;

         const socket: TypedSocket = io(url, {
           query: projectId ? { projectId } : undefined,
           autoConnect,
           reconnection: true,
           reconnectionAttempts: Infinity,
           reconnectionDelay: 1000,
           reconnectionDelayMax: 5000,
         });

         return socket;
       }

       /**
        * Telemetry callback for render performance monitoring.
        * frameTime: ms taken to render buffered tokens
        * tokenCount: number of tokens in the batch
        */
       export type RenderTelemetry = (frameTime: number, tokenCount: number) => void;

       /**
        * Token buffer state for requestAnimationFrame batching.
        * Prevents UI backpressure per CONTEXT.md INFRA-02.
        */
       export interface TokenBufferState {
         tokens: string[];
         subscribe: (agentId: string) => void;
         unsubscribe: (agentId: string) => void;
         clear: () => void;
       }

       /**
        * Create RAF-based token buffer.
        *
        * @param socket - Typed Socket.IO client
        * @param onUpdate - Callback when buffer flushes to state
        * @param onTelemetry - Optional telemetry callback for slow-frame detection
        * @param maxBufferSize - Cap buffer size to prevent memory leaks (default 1000)
        * @returns Buffer control functions
        */
       export function createTokenBuffer(
         socket: TypedSocket,
         onUpdate: (tokens: string[]) => void,
         onTelemetry?: RenderTelemetry,
         maxBufferSize = 1000
       ): TokenBufferState {
         let buffer: string[] = [];
         let tokens: string[] = [];
         let frameId: number | null = null;
         let currentAgentId: string | null = null;

         function flush(): void {
           if (buffer.length > 0) {
             const frameStart = performance.now();
             const batchSize = buffer.length;

             // Flush buffer to tokens state
             tokens = [...tokens, ...buffer];
             buffer = [];

             // Notify consumer
             onUpdate(tokens);

             // Collect telemetry for v3.0 adaptive throttling decision
             if (onTelemetry) {
               // Measure after render settles
               requestAnimationFrame(() => {
                 const frameTime = performance.now() - frameStart;
                 onTelemetry(frameTime, batchSize);
               });
             }
           }

           // Continue RAF loop
           frameId = requestAnimationFrame(flush);
         }

         function handleToken(event: TokenEvent): void {
           if (event.agentId !== currentAgentId) return;

           // Cap buffer to prevent memory leak per RESEARCH.md pitfall
           if (buffer.length >= maxBufferSize) {
             buffer.shift();  // Drop oldest
           }
           buffer.push(event.token);
         }

         function subscribe(agentId: string): void {
           currentAgentId = agentId;
           socket.emit('agent:subscribe', agentId);
           socket.on('agent:token', handleToken);
           frameId = requestAnimationFrame(flush);
         }

         function unsubscribe(agentId: string): void {
           socket.off('agent:token', handleToken);
           if (frameId !== null) {
             cancelAnimationFrame(frameId);
             frameId = null;
           }
           currentAgentId = null;
         }

         function clear(): void {
           buffer = [];
           tokens = [];
           onUpdate(tokens);
         }

         return {
           get tokens() { return tokens; },
           subscribe,
           unsubscribe,
           clear,
         };
       }
       ```

       ```typescript
       import type { Server } from 'socket.io';
       import type { HealthMetricsEvent } from '@gsd/events';

       const startTime = Date.now();

       export function collectHealthMetrics(io: Server): HealthMetricsEvent {
         const sockets = io.sockets.sockets;
         const connectedClients = sockets.size;

         // Collect room counts
         const roomCounts: Record<string, number> = {};
         const rooms = io.sockets.adapter.rooms;
         rooms.forEach((members, roomName) => {
           // Skip socket ID rooms (each socket auto-joins its own ID room)
           if (!sockets.has(roomName)) {
             roomCounts[roomName] = members.size;
           }
         });

         return {
           connectedClients,
           roomCounts,
           uptime: Date.now() - startTime,
           memoryUsage: process.memoryUsage().heapUsed,
         };
       }

       /**
        * Start periodic health broadcast.
        * @param io Socket.IO server
        * @param intervalMs Broadcast interval (default 30000)
        */
       export function startHealthBroadcast(io: Server, intervalMs = 30000): () => void {
         const timer = setInterval(() => {
           const metrics = collectHealthMetrics(io);
           io.emit('connection:health', metrics);
         }, intervalMs);

         return () => clearInterval(timer);
       }
       ```
